```dart
/// Two-mode phonetic bridge.
/// - SimplePronunciation: plain string ("a-bu-si-a").
/// - SegmentedPronunciation: structured segments for homophone disambiguation
///   (bo vs bɔ). Graphemes are exact; equality NEVER folds ɔ→o or ɛ→e.
enum Tone { high, low, mid, rising, falling }
enum AtrClass { advanced, retracted } // +ATR (o,e,i,u) vs -ATR (ɔ,ɛ)

class PhoneticSegment {
  final String orthography; // exact graphemes: 'ɔ' != 'o'
  final String ipa;
  final Tone tone;
  final AtrClass atr;
  final bool nasalized;
  const PhoneticSegment({
    required this.orthography,
    required this.ipa,
    required this.tone,
    required this.atr,
    this.nasalized = false,
  });

  @override
  bool operator ==(Object o) =>
      o is PhoneticSegment &&
      o.orthography == orthography && // preserves bo/bɔ distinction
      o.tone == tone && o.atr == atr && o.nasalized == nasalized;

  @override
  int get hashCode => Object.hash(orthography, tone, atr, nasalized);
}

sealed class PhoneticBridge {
  const PhoneticBridge();
  String get pronunciation; // simple string always available
}

class SimplePronunciation extends PhoneticBridge {
  @override
  final String pronunciation;
  const SimplePronunciation(this.pronunciation);

  @override
  bool operator ==(Object o) =>
      o is SimplePronunciation && o.pronunciation == pronunciation;
  @override
  int get hashCode => pronunciation.hashCode;
}

class SegmentedPronunciation extends PhoneticBridge {
  @override
  final String pronunciation;
  final List<PhoneticSegment> segments;
  final String homophoneSetId; // groups TRUE homophones; bo/bɔ differ → different ids
  final String? senseKey;
  const SegmentedPronunciation({
    required this.pronunciation,
    required this.segments,
    required this.homophoneSetId,
    this.senseKey,
  });

  @override
  bool operator ==(Object o) =>
      o is SegmentedPronunciation &&
      o.pronunciation == pronunciation &&
      o.homophoneSetId == homophoneSetId &&
      o.senseKey == senseKey &&
      _listEq(o.segments, segments);
  @override
  int get hashCode => Object.hash(pronunciation, homophoneSetId, senseKey, Object.hashAll(segments));
}

bool _listEq(List a, List b) {
  if (a.length != b.length) return false;
  for (var i = 0; i < a.length; i++) {
    if (a[i] != b[i]) return false;
  }
  return true;
}
```

```dart
/// Rule 3 — ALL JSON parsing lives in data/. UI never sees JSON.
class CurriculumUnitDto {
  final Map<String, dynamic> json;
  const CurriculumUnitDto(this.json);

  factory CurriculumUnitDto.fromJson(Map<String, dynamic> j) => CurriculumUnitDto(j);

  CurriculumUnit toEntity() {
    _requireKeys(json, const [
      'unit_title', 'vocabulary_spotlight', 'grammar_mechanics',
      'lineage_challenges', 'theological_framework',
    ]); // I1

    final rawChallenges = (json['lineage_challenges'] as List?) ?? const [];
    final challenges = rawChallenges
        .map((c) => _challenge(c as Map<String, dynamic>))
        .toList();

    // NOTE: json['action_button'] is intentionally NOT read (I3).
    return CurriculumUnit(
      unitTitle: json['unit_title'] as String,
      vocabularySpotlight: _vocab(json['vocabulary_spotlight'] as Map<String, dynamic>),
      grammarMechanics: _grammar(json['grammar_mechanics'] as Map<String, dynamic>),
      lineageChallenges: LineageChallengeSet(challenges), // throws if != 10 (I2)
      theologicalFramework: TheologicalFramework.fromKey(json['theological_framework'] as String),
    );
  }

  static void _requireKeys(Map<String, dynamic> j, List<String> keys) {
    for (final k in keys) {
      if (!j.containsKey(k)) throw SchemaFailure('Missing required key "$k"');
    }
  }

  static PhoneticBridge _phonetics(Map<String, dynamic> p) {
    final pronunciation = p['pronunciation'] as String;
    if (p['segments'] is List) {
      final segs = (p['segments'] as List).map((s) {
        final m = s as Map<String, dynamic>;
        return PhoneticSegment(
          orthography: m['orthography'] as String, // exact graphemes preserved
          ipa: m['ipa'] as String,
          tone: Tone.values.byName(m['tone'] as String),
          atr: AtrClass.values.byName(m['atr'] as String),
          nasalized: (m['nasalized'] as bool?) ?? false,
        );
      }).toList();
      return SegmentedPronunciation(
        pronunciation: pronunciation,
        segments: segs,
        homophoneSetId: p['homophone_set_id'] as String,
        senseKey: p['sense_key'] as String?,
      );
    }
    return SimplePronunciation(pronunciation);
  }

  static VocabularySpotlight _vocab(Map<String, dynamic> v) => VocabularySpotlight(
        headword: v['headword'] as String,
        gloss: v['gloss'] as String,
        phonetics: _phonetics(v['phonetic_bridge'] as Map<String, dynamic>),
        theologicalFramework: TheologicalFramework.fromKey(v['theological_framework'] as String),
        exampleSentences: ((v['example_sentences'] as List?) ?? const []).cast<String>(),
      );

  static GrammarMechanics _grammar(Map<String, dynamic> g) => GrammarMechanics(
        focus: g['focus'] as String,
        explanation: g['explanation'] as String,
        patterns: ((g['patterns'] as List?) ?? const []).cast<String>(),
      );

  static LineageChallenge _challenge(Map<String, dynamic> c) => LineageChallenge(
        prompt: c['prompt'] as String,
        type: _type(c['type'] as String),
        options: (c['options'] as List).cast<String>(),
        correctIndex: c['correct_index'] as int,
        phoneticTarget: c['phonetic_target'] == null
            ? null
            : _phonetics(c['phonetic_target'] as Map<String, dynamic>),
        theologicalFramework: TheologicalFramework.fromKey(c['theological_framework'] as String),
      );

  static ChallengeType _type(String t) => switch (t) {
        'recall' => ChallengeType.recall,
        'phonetic_discrimination' => ChallengeType.phoneticDiscrimination,
        'lineage_ordering' => ChallengeType.lineageOrdering,
        'concept_match' => ChallengeType.conceptMatch,
        _ => throw SchemaFailure('Unknown challenge type: $t'),
      };
}
```

```dart
/// I4/I5 — rejects any unit whose text uses diaspora-mapping or non-Akan analogies.
/// Mirrors scripts/validate_curriculum.mjs so client and CI agree.
class CulturalIntegrityGuard {
  static final List<RegExp> _banned = [
    RegExp(r'\bsimilar to\b', caseSensitive: false),
    RegExp(r'\bequivalent of\b', caseSensitive: false),
    RegExp(r'\bakin to\b', caseSensitive: false),
    RegExp(r'\bcomparable to\b', caseSensitive: false),
    RegExp(r'\bdiaspora\b', caseSensitive: false),
    RegExp(r'\bwestern\b', caseSensitive: false),
    RegExp(r'\bchristian\b', caseSensitive: false),
    RegExp(r'\bbiblical\b', caseSensitive: false),
    RegExp(r'\bgreek\b', caseSensitive: false),
    RegExp(r'\broman\b', caseSensitive: false),
  ];

  /// Returns null when clean, or an IntegrityFailure describing the first hit.
  static IntegrityFailure? scan(String text) {
    for (final re in _banned) {
      final m = re.firstMatch(text);
      if (m != null) return IntegrityFailure('Banned analogy/diaspora term: "${m.group(0)}"');
    }
    return null;
  }
}
```

```ts
import * as functions from "firebase-functions/v2/firestore";

// Server-side re-validation of I1–I3, I6 on every write (clients are bypassable).
export const validateCurriculum = functions.onDocumentWritten(
  "curriculum_units/{id}",
  async (event) => {
    const data = event.data?.after.data();
    if (!data) return;
    const errors: string[] = [];
    for (const k of ["unit_title","vocabulary_spotlight","grammar_mechanics","lineage_challenges","theological_framework"]) {
      if (!(k in data)) errors.push(`I1 missing ${k}`);
    }
    if (!Array.isArray(data.lineage_challenges) || data.lineage_challenges.length !== 10) errors.push("I2 not exactly 10");
    if ("action_button" in data) errors.push("I3 action_button present");
    if (errors.length) {
      await event.data?.after.ref.delete(); // reject non-conforming write
      console.error("Rejected curriculum_units doc:", errors);
    }
  }
);
```
